### A Quick Code Example

Here's a simple C++ snippet to get things started:

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    std::vector<int> numbers = {5, 2, 8, 1, 9};
    std::sort(numbers.begin(), numbers.end());

    for (const auto& n : numbers) {
        std::cout << n << " ";
    }
    return 0;
}
```
